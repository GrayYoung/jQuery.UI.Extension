jQuery UI Zoom

	$('#destroy').click(function() {
		$('#zoom').zoom('destroy');			
	});
	
	$('#rebuild').click(function() {
		$('#zoom').zoom();		
	}).click();		
	
	$('#standard').click(function() {
		$('#zoom').zoom({
			width : 100,
			height : 100,
			times : 3,
			type : 'standard'
		});		
	});
	
	$('#inner').click(function() {
		$('#zoom').zoom({
			type : 'inner',
			width : 100
		});	
	});
	
	$('#draggable').click(function() {
		$('#zoom').zoom({
			type : 'drag'
		});		
	});
	
	$('#alwaysOn').click(function() {
		$('#zoom').zoom('option', {
			alwaysOn : true	
		});
	});
